# Get Sessions

Retrieve information about a specific session or list all sessions with optional filtering.

## Get a Specific Session

Retrieve detailed information about a single session by ID.

```
GET https://api.voicevora.com/v1/sessions/:id
```

### Path Parameters

| Parameter | Type | Description |
|-----------|------|-------------|
| `id` | string | The session ID (e.g., `session_abc123`) |

## List All Sessions

Retrieve a paginated list of sessions with optional filtering.

```
GET https://api.voicevora.com/v1/sessions
```

### Query Parameters

| Parameter | Type | Default | Description |
|-----------|------|---------|-------------|
| `limit` | integer | `20` | Number of sessions per page (max: 100) |
| `cursor` | string | - | Pagination cursor |
| `agent_id` | string | - | Filter by agent ID |
| `status` | string | - | Filter by status |
| `started_after` | string | - | Filter by start date (ISO 8601) |
| `started_before` | string | - | Filter by start date (ISO 8601) |
| `test_mode` | boolean | - | Filter by test mode |
| `order` | string | `desc` | Sort order: `asc` or `desc` |
| `order_by` | string | `created_at` | Sort field |

#### Paginate Through All Sessions

<CodeGroup>
```javascript JavaScript
async function* getAllSessions(filters = {}) {
  let cursor = null;

  do {
    const response = await vora.sessions.list({
      ...filters,
      limit: 100,
      cursor
    });

    for (const session of response.data) {
      yield session;
    }

    cursor = response.pagination.next_cursor;
  } while (cursor);
}

// Process all sessions
let totalDuration = 0;
for await (const session of getAllSessions({ status: 'completed' })) {
  totalDuration += session.duration_seconds;
}

console.log(`Total talk time: ${totalDuration} seconds`);
```

```python Python
def get_all_sessions(**filters):
    cursor = None

    while True:
        response = vora.sessions.list(limit=100, cursor=cursor, **filters)

        for session in response.data:
            yield session

        cursor = response.pagination.next_cursor
        if not cursor:
            break

# Process all sessions
total_duration = sum(
    s.duration_seconds
    for s in get_all_sessions(status='completed')
)
print(f"Total talk time: {total_duration} seconds")
```
</CodeGroup>

## Monitoring Active Sessions

Track currently active sessions for your dashboard:

<CodeGroup>
```javascript JavaScript
async function monitorActiveSessions() {
  const { data: active, pagination } = await vora.sessions.list({
    status: 'active',
    limit: 100
  });

  console.log(`Active sessions: ${pagination.total}`);

  active.forEach(session => {
    const duration = Math.floor(
      (Date.now() - new Date(session.started_at).getTime()) / 1000
    );
    console.log(`  ${session.id}: ${duration}s (${session.participant?.name || 'Anonymous'})`);
  });

  return active;
}

// Poll every 5 seconds
setInterval(monitorActiveSessions, 5000);
```

```python Python
import time
from datetime import datetime

def monitor_active_sessions():
    result = vora.sessions.list(status='active', limit=100)

    print(f"Active sessions: {result.pagination.total}")

    for session in result.data:
        started = datetime.fromisoformat(session.started_at.replace('Z', '+00:00'))
        duration = int((datetime.now(started.tzinfo) - started).total_seconds())
        name = session.participant.get('name', 'Anonymous') if session.participant else 'Anonymous'
        print(f"  {session.id}: {duration}s ({name})")

    return result.data

# Poll every 5 seconds
while True:
    monitor_active_sessions()
    time.sleep(5)
```
</CodeGroup>

## Session Analytics

Aggregate session data for analytics:

<CodeGroup>
```javascript JavaScript
async function getSessionAnalytics(agentId, startDate, endDate) {
  const sessions = [];

  for await (const session of getAllSessions({
    agent_id: agentId,
    status: 'completed',
    started_after: startDate,
    started_before: endDate
  })) {
    sessions.push(session);
  }

  const analytics = {
    totalSessions: sessions.length,
    totalDuration: sessions.reduce((sum, s) => sum + s.duration_seconds, 0),
    avgDuration: sessions.length > 0
      ? sessions.reduce((sum, s) => sum + s.duration_seconds, 0) / sessions.length
      : 0,
    avgTurns: sessions.length > 0
      ? sessions.reduce((sum, s) => sum + s.metrics.turns, 0) / sessions.length
      : 0,
    avgSentiment: sessions.length > 0
      ? sessions.reduce((sum, s) => sum + s.metrics.sentiment_score, 0) / sessions.length
      : 0
  };

  return analytics;
}

const stats = await getSessionAnalytics(
  'agent_abc123xyz',
  '2024-01-01T00:00:00Z',
  '2024-01-31T23:59:59Z'
);

console.log(`Sessions: ${stats.totalSessions}`);
console.log(`Avg duration: ${Math.round(stats.avgDuration)}s`);
console.log(`Avg sentiment: ${stats.avgSentiment.toFixed(2)}`);
```

```python Python
def get_session_analytics(agent_id, start_date, end_date):
    sessions = list(get_all_sessions(
        agent_id=agent_id,
        status='completed',
        started_after=start_date,
        started_before=end_date
    ))

    if not sessions:
        return {
            'total_sessions': 0,
            'total_duration': 0,
            'avg_duration': 0,
            'avg_turns': 0,
            'avg_sentiment': 0
        }

    return {
        'total_sessions': len(sessions),
        'total_duration': sum(s.duration_seconds for s in sessions),
        'avg_duration': sum(s.duration_seconds for s in sessions) / len(sessions),
        'avg_turns': sum(s.metrics['turns'] for s in sessions) / len(sessions),
        'avg_sentiment': sum(s.metrics['sentiment_score'] for s in sessions) / len(sessions)
    }

stats = get_session_analytics(
    'agent_abc123xyz',
    '2024-01-01T00:00:00Z',
    '2024-01-31T23:59:59Z'
)

print(f"Sessions: {stats['total_sessions']}")
print(f"Avg duration: {round(stats['avg_duration'])}s")
print(f"Avg sentiment: {stats['avg_sentiment']:.2f}")
```
