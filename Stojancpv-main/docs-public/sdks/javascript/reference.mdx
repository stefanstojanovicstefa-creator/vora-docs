# JavaScript SDK Reference

Complete reference for all methods and types in the Vora JavaScript SDK.

---

## VoraClient

Main client class for API access.

### Constructor

```typescript
import { VoraClient } from '@vora-ai/sdk';

const vora = new VoraClient(options: VoraClientOptions);
```

### Options

```typescript
interface VoraClientOptions {
  apiKey: string;                // Required: Your API key
  baseUrl?: string;              // Custom API endpoint
  timeout?: number;              // Request timeout (ms)
  maxRetries?: number;           // Retry attempts
  retryDelay?: number;           // Base retry delay (ms)
  fetch?: typeof fetch;          // Custom fetch implementation
}
```

## Agents

### agents.create

Create a new agent.

```typescript
const agent = await vora.agents.create(input: CreateAgentInput): Promise<Agent>
```

**Input:**

```typescript
interface CreateAgentInput {
  name: string;
  systemPrompt: string;
  voice?: VoiceConfig;
  model?: ModelConfig;
  greeting?: string;
  endConversationPhrases?: string[];
  functions?: FunctionDefinition[];
  knowledgeBase?: KnowledgeBaseConfig;
  metadata?: Record<string, any>;
}
```

### agents.get

Get an agent by ID.

```typescript
const agent = await vora.agents.get(
  id: string,
  options?: { include?: string[] }
): Promise<Agent>
```

### agents.list

List all agents.

```typescript
const result = await vora.agents.list(options?: ListOptions): Promise<ListResult<Agent>>
```

**Options:**

```typescript
interface ListOptions {
  limit?: number;                // Default: 20, Max: 100
  after?: string;                // Cursor for pagination
  status?: AgentStatus;          // Filter by status
  orderBy?: 'createdAt' | 'updatedAt' | 'name';
  order?: 'asc' | 'desc';
}
```

### agents.update

Update an agent.

```typescript
const agent = await vora.agents.update(
  id: string,
  input: UpdateAgentInput
): Promise<Agent>
```

### agents.delete

Delete an agent.

```typescript
await vora.agents.delete(id: string): Promise<void>
```

### agents.pause

Pause an agent.

```typescript
const agent = await vora.agents.pause(id: string): Promise<Agent>
```

### agents.resume

Resume a paused agent.

```typescript
const agent = await vora.agents.resume(id: string): Promise<Agent>
```

### agents.clone

Clone an agent.

```typescript
const clone = await vora.agents.clone(
  id: string,
  options?: { name?: string; status?: AgentStatus }
): Promise<Agent>
```

### agents.test

Test an agent.

```typescript
const result = await vora.agents.test(
  id: string,
  input: { message: string; variables?: Record<string, any> }
): Promise<{ reply: string; latency: number }>
```

### sessions.get

### sessions.list


### sessions.update

```typescript

  id: string,
  input: { metadata?: Record<string, any> }

```

### sessions.end

```typescript

```

### sessions.getTranscript

Get session transcript.

### sessions.getRecording

### sessions.transfer

Transfer a session.

## Types

### Agent

```typescript
interface Agent {
  id: string;
  name: string;
  systemPrompt: string;
  voice?: VoiceConfig;
  model?: ModelConfig;
  greeting?: string;
  endConversationPhrases?: string[];
  functions?: FunctionDefinition[];
  knowledgeBase?: KnowledgeBaseConfig;
  status: AgentStatus;
  metadata?: Record<string, any>;
  createdAt: Date;
  updatedAt: Date;
}

type AgentStatus = 'draft' | 'active' | 'paused' | 'archived';
```

### Session

```typescript
interface Session {
  id: string;
  agentId: string;
  userId?: string;
  status: SessionStatus;
  token: string;
  roomUrl: string;
  duration?: number;
  variables?: Record<string, any>;
  metadata?: Record<string, any>;
  expiresAt: Date;
  createdAt: Date;
  endedAt?: Date;
}

type SessionStatus = 'created' | 'connecting' | 'active' | 'ending' | 'completed' | 'failed';
```

### VoiceConfig

```typescript
interface VoiceConfig {
  provider: 'elevenlabs' | 'cartesia' | 'google';
  voiceId?: string;
  model?: string;
  stability?: number;
  similarityBoost?: number;
  style?: number;
  useSpeakerBoost?: boolean;
  speed?: number;
  languageCode?: string;
  voiceName?: string;
  speakingRate?: number;
  pitch?: number;
}
```

### ModelConfig

```typescript
interface ModelConfig {
  provider: 'openai' | 'anthropic' | 'google' | 'groq';
  model: string;
  temperature?: number;
  maxTokens?: number;
  topP?: number;
  frequencyPenalty?: number;
  presencePenalty?: number;
}
```

### FunctionDefinition

```typescript
interface FunctionDefinition {
  name: string;
  description: string;
  parameters: JSONSchema;
  handler: FunctionHandler;
}

interface FunctionHandler {
  type: 'webhook' | 'builtin';
  url?: string;
  method?: 'GET' | 'POST' | 'PUT' | 'DELETE';
  headers?: Record<string, string>;
  action?: string;
  config?: Record<string, any>;
}
```

### Transcript

```typescript
interface Transcript {
  sessionId: string;
  messages: TranscriptMessage[];
  duration: number;
  wordCount: number;
}

interface TranscriptMessage {
  speaker: 'agent' | 'user';
  text: string;
  timestamp: Date;
}
```

## Errors

```typescript
import {
  VoraError,              // Base error class
  AuthenticationError,    // 401 errors
  AuthorizationError,     // 403 errors
  NotFoundError,          // 404 errors
  ValidationError,        // 400 errors
  RateLimitError,         // 429 errors
  InternalError           // 500 errors
} from '@vora-ai/sdk';

// Error properties
interface VoraError extends Error {
  type: string;
  code: string;
  message: string;
  details?: Record<string, any>;
  requestId?: string;
}

interface RateLimitError extends VoraError {
  retryAfter: number;     // Seconds until retry
}
```
