## Configuration

### Basic Setup

```typescript
const agent = await vora.agents.create({
  name: 'Multilingual Agent',
  systemPrompt: 'You are a helpful assistant.',
  voice: {
    provider: 'google',
    languageCode: 'en-US',
    name: 'en-US-Neural2-F'
  }
});
```

### Full Configuration

```typescript
const agent = await vora.agents.create({
  name: 'Custom Voice Agent',
  systemPrompt: 'You are a professional assistant.',
  voice: {
    provider: 'google',
    languageCode: 'en-US',
    name: 'en-US-Neural2-F',
    speakingRate: 1.0,
    pitch: 0.0,
    volumeGainDb: 0.0,
    effectsProfileId: ['telephony-class-application']
  }
});
```

## Voice Settings

### Speaking Rate

Control speech speed:

```typescript
voice: {
  provider: 'google',
  speakingRate: 1.0  // 0.25 - 4.0
}
```

| Value | Effect |
|-------|--------|
| 0.5 | Half speed |
| 0.75 | Slow |
| 1.0 | Normal |
| 1.25 | Fast |
| 1.5 | Very fast |

### Pitch

Adjust voice pitch:

```typescript
voice: {
  provider: 'google',
  pitch: 0.0  // -20.0 to 20.0 semitones
}
```

| Value | Effect |
|-------|--------|
| -10 | Much lower |
| -5 | Lower |
| 0 | Normal |
| 5 | Higher |
| 10 | Much higher |

### Volume

Adjust output volume:

```typescript
voice: {
  provider: 'google',
  volumeGainDb: 0.0  // -96.0 to 16.0 dB
}
```

### Audio Effects

Optimize for specific playback:

```typescript
voice: {
  provider: 'google',
  effectsProfileId: ['telephony-class-application']
}
```

| Profile | Use Case |
|---------|----------|
| `telephony-class-application` | Phone systems |
| `wearable-class-device` | Smartwatches |
| `headphone-class-device` | Headphones |
| `small-bluetooth-speaker-class-device` | Bluetooth speakers |
| `large-home-entertainment-class-device` | Home speakers |

```typescript

```

### Phone Systems (IVR)

```typescript
voice: {
  provider: 'google',
  languageCode: 'en-US',
  name: 'en-US-Neural2-F',
  effectsProfileId: ['telephony-class-application'],
  speakingRate: 0.95

```

### Multilingual Agents

```typescript
// Support multiple languages with one agent
voice: {
  provider: 'google',
  languageCode: userLanguage,
  name: getVoiceForLanguage(userLanguage)
}

function getVoiceForLanguage(lang: string) {
  const voices = {
    'en-US': 'en-US-Neural2-F',
    'es-ES': 'es-ES-Neural2-A',
    'fr-FR': 'fr-FR-Neural2-A',
    'de-DE': 'de-DE-Neural2-A',
    'ja-JP': 'ja-JP-Neural2-B'
  };
  return voices[lang] || 'en-US-Neural2-F';
}
```

    Create a language-aware voice selector:

    ```typescript
    const voiceMap = {
      'en': { name: 'en-US-Neural2-F', code: 'en-US' },
      'es': { name: 'es-ES-Neural2-A', code: 'es-ES' },
      'fr': { name: 'fr-FR-Neural2-A', code: 'fr-FR' },
      // Add more as needed
    };

    function getVoice(langCode: string) {
      const lang = langCode.split('-')[0];
      return voiceMap[lang] || voiceMap['en'];
    }
    ```
  </Accordion>

  <Accordion title="Use SSML for Control">
    For precise pronunciation:

    ```xml
    <speak>

    </speak>
    ```

## API Key Setup

### Service Account (Recommended)

```bash
# Set path to credentials file
GOOGLE_APPLICATION_CREDENTIALS=/path/to/service-account.json
```

### Get Credentials

1. Go to [Google Cloud Console](https://console.cloud.google.com)
2. Create or select a project
3. Enable Text-to-Speech API
4. Create a service account
5. Download JSON key file
6. Set environment variable

### Alternative: API Key

For simpler setup (less secure):

```bash
GOOGLE_API_KEY=AIza...
```

## Troubleshooting

<AccordionGroup>
  <Accordion title="Authentication Errors">
    If authentication fails:

    - Verify service account JSON exists
    - Check GOOGLE_APPLICATION_CREDENTIALS path
    - Ensure Text-to-Speech API is enabled
    - Verify project billing is active
  </Accordion>

  <Accordion title="Voice Not Found">
    If voice name is invalid:

    - Check exact voice name format
    - Verify language code matches
    - Use Neural2 voices (most reliable)
    - Check voice availability in your region
  </Accordion>

  <Accordion title="Audio Quality Issues">
    If audio sounds robotic:

    - Upgrade from Standard to Neural2
    - Check effectsProfileId matches use case
    - Adjust speakingRate and pitch
    - Consider Studio voices for premium quality
  </Accordion>
</AccordionGroup>
